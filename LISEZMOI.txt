└───────────────────────────────────────────────────────────────────────────────┘

  Créer une vente en PHP :

  <?php
  require_once 'models/Vente.php';
  
  $venteModel = new Vente();
  
  $venteId = $venteModel->createVente([
      'client_id' => 1,
      'entrepot_id' => 1,
      'user_id' => 1,
      'date_vente' => date('Y-m-d'),
      'status' => 'confirmee'
  ], [
      [
          'product_id' => 1,
          'quantite' => 2,
          'prix_unitaire' => 50000,
          'tva' => 18,
          'sous_total' => 100000
      ]
  ]);
  
  // ✅ Vente créée
  // ✅ Stock mis à jour automatiquement
  // ✅ Mouvement de stock créé et validé
  ?>
